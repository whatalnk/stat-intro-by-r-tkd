my.t.test <- function(x, y){
    n.x <- length(x)
    n.y <- length(x)
    
    mean.x <- mean(x)
    mean.y <- mean(y)
    ss.x <- sum((x - mean.x)^2)
    ss.y <- sum((y - mean.y)^2)
    sp2 <- (ss.x + ss.y) / (n.x + n.y - 2)
    se_mean <- sqrt(sp2 * (n.x + n.y) / (n.x * n.y))
    t_val <- (mean.x - mean.y) / se_mean
    return(t_val)
}
