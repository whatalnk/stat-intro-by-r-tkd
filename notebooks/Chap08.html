ピアソンの積率相関係数$r$

$$ r = \frac{SP_{XY}}{\sqrt{SS_{X} SS_{Y}}} = \frac{\sum\limits^n_{i=1}(X_i - \bar{X})(Y_i - \bar{Y})}{\sqrt{\sum\limits^n_{i=1}(X_i - \bar{X})^2 \sum\limits^n_{i=1}(Y_i - \bar{Y})^2}} $$

my_cor <- function(x, y){
    sp.xy <- sum((x - mean(x)) * (y - mean(y)))
    sp.x <- sum((x - mean(x))**2)
    sp.y <- sum((y - mean(y))**2)
    return(sp.xy / sqrt(sp.x * sp.y))
}
